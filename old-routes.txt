Route::get('/registration/success', function () {
    // محاولة الحصول على البيانات من مصادر متعددة
    $studentData = null;

    // 1. جلب من session
    $studentData = session()->get('student_data');

    // 2. إذا لم توجد في session، تحقق من flash data
    if (! $studentData) {
        $studentData = [
            'student_id' => session('student_id') ?? session()->get('student_id') ?? 'N/A',
            'student_name' => 'مستخدم جديد',
            'email' => session('email') ?? 'غير محدد',
            'phone' => 'غير محدد',
            'date_of_birth' => 'غير محدد',
            'gender' => 'غير محدد',
            'education_level' => 'غير محدد',
            'parent_phone' => 'غير محدد',
            'interests' => 'غير محدد',
            'how_know_us' => 'غير محدد',
            'username' => 'سيتم تعيينه',
            'course_name' => 'سيتم تحديده من قبل الإدارة',
            'registration_date' => now()->format('Y/m-d h:i A'),
        ];

        // 3. إذا كان هناك student_id محدد، حاول جلب من قاعدة البيانات
        $studentId = request()->get('student_id') ?? session('student_id');
        if ($studentId && $studentId !== 'N/A') {
            try {
                $student = \App\Models\Student::with(['user', 'user.profile', 'preferredCourse'])->find($studentId);

                if ($student) {
                    $studentData = [
                        'student_id' => $student->student_id,
                        'student_name' => $student->student_name,
                        'username' => $student->user ? $student->user->username : 'غير محدد',
                        'email' => $student->user ? $student->user->email : 'غير محدد',
                        'phone' => $student->user && $student->user->profile ?
                            $student->user->profile->phone_number : 'غير محدد',
                        'date_of_birth' => $student->user && $student->user->profile ?
                            $student->user->profile->date_of_birth : 'غير محدد',
                        'gender' => $student->user && $student->user->profile ?
                            $student->user->profile->gender : 'غير محدد',
                        'education_level' => $student->user && $student->user->profile ?
                            $student->user->profile->education_level : 'غير محدد',
                        'parent_phone' => $student->user && $student->user->profile ?
                            $student->user->profile->parent_phone : 'غير محدد',
                        'interests' => $student->user && $student->user->profile ?
                            $student->user->profile->interests : 'غير محدد',
                        'how_know_us' => $student->user && $student->user->profile ?
                            $student->user->profile->how_know_us : 'غير محدد',
                        'course_id' => $student->preferred_course_id,
                        'course_name' => $student->preferredCourse ?
                            $student->preferredCourse->course_name : 'غير محدد',
                        'registration_date' => $student->user ?
                            $student->user->created_at->format('Y/m-d h:i A') : now()->format('Y/m-d h:i A'),
                    ];
                }
            } catch (\Exception $e) {
                \Log::error('Error fetching student data: '.$e->getMessage());
            }
        }
    }

    // 4. التأكد من وجود student_id
    if (! isset($studentData['student_id'])) {
        $studentData['student_id'] = 'N/A';
    }

    // 5. تسجيل لل debugging
    \Log::info('Registration success page data:', [
        'has_student_data' => ! empty($studentData),
        'student_id' => $studentData['student_id'] ?? 'N/A',
        'student_name' => $studentData['student_name'] ?? 'N/A',
        'all_keys' => array_keys($studentData ?? []),
    ]);

    return view('auth.registration-success', compact('studentData'));
})->name('registration.success');

    Route::get('/students/{student}/payment-status', function ($studentId) {
        try {
            $student = \App\Models\Student::with(['invoices'])->find($studentId);

            if (! $student) {
                return response()->json([
                    'success' => false,
                    'message' => 'Student not found',
                ]);
            }

            $totalPaid = $student->invoices->sum('amount_paid');
            $totalRequired = $student->invoices->sum('amount');
            $hasDebt = $totalPaid < $totalRequired;

            return response()->json([
                'success' => true,
                'student_id' => $studentId,
                'total_paid' => $totalPaid,
                'total_required' => $totalRequired,
                'has_debt' => $hasDebt,
                'balance' => $totalRequired - $totalPaid,
                'invoices_count' => $student->invoices->count(),
                'invoices' => $student->invoices->map(function ($invoice) {
                    return [
                        'id' => $invoice->invoice_id,
                        'amount' => $invoice->amount,
                        'amount_paid' => $invoice->amount_paid,
                        'status' => $invoice->status,
                        'due_date' => $invoice->due_date,
                    ];
                }),
            ]);
        } catch (\Exception $e) {
            \Log::error('Error checking payment status: '.$e->getMessage());

            return response()->json([
                'success' => false,
                'message' => 'Server error',
            ], 500);
        }
    })->name('students.payment-status');

        try {
            $groupInfo = \App\Models\WaitingGroup::where('group_name', $groupName)
                ->selectRaw('
                group_name,
                COUNT(*) as student_count,
                MAX(created_at) as created_at,
                MAX(course_id) as course_id
            ')
                ->groupBy('group_name')
                ->first();

            if (! $groupInfo) {
                return response()->json([
                    'success' => false,
                    'message' => 'المجموعة غير موجودة',
                ]);
            }

            $course = \App\Models\Course::find($groupInfo->course_id);

            return response()->json([
                'success' => true,
                'group_name' => $groupInfo->group_name,
                'student_count' => $groupInfo->student_count,
                'course_name' => $course ? $course->course_name : null,
                'created_at' => \Carbon\Carbon::parse($groupInfo->created_at)->format('d/m/Y'),
            ]);

        } catch (\Exception $e) {
            return response()->json([
                'success' => false,
                'message' => 'حدث خطأ في جلب المعلومات',

    Route::get('/students/{student}/check-invoices', function ($student) {
        try {
            // Check if student exists
            $studentRecord = \App\Models\Student::find($student);

            if (! $studentRecord) {
                return response()->json([
                    'has_unpaid_invoices' => false,
                    'error' => 'Student not found',
                ], 404);
            }

            // Check for unpaid invoices
            $hasUnpaidInvoices = \App\Models\Invoice::where('student_id', $student)
                ->where(function ($query) {
                    $query->where('status', '!=', 'paid')
                        ->orWhere('amount_paid', '<', DB::raw('amount'));
                })
                ->exists();

            return response()->json([
                'student_id' => $student,
                'student_name' => $studentRecord->student_name,
                'has_unpaid_invoices' => $hasUnpaidInvoices,
            ]);
        } catch (\Exception $e) {
            \Log::error('Error checking student invoices: '.$e->getMessage());

            return response()->json([
                'has_unpaid_invoices' => false,
                'error' => 'Server error',
            ], 500);
        }
    })->name('students.check-invoices');

    Route::post('/mark_notifications_read', function () {
        try {
            $userId = auth()->id();

            if (! $userId) {
                return response()->json(['success' => false, 'message' => 'Not authenticated'], 401);
            }

            $updated = DB::table('notifications')
                ->where('user_id', $userId)
                ->update(['is_read' => 1]);

            return response()->json([
                'success' => true,
                'message' => 'Notifications marked as read',
                'updated_count' => $updated,
            ]);
        } catch (\Exception $e) {
            \Log::error('Error marking notifications read: '.$e->getMessage());

            return response()->json(['success' => false, 'message' => 'Server error'], 500);
        }
    })->name('notifications.markRead');
